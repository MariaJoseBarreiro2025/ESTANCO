INICIALIZACIÓN DE GIT

# Listar archivos ocultos
ls -a

# Listar con detalles
ls -la

PRIMER COMMIT - ARCHIVO INICIAL

# Ver estado inicial
git status

# Añadir archivo al staging
git add "marcas tabaco.txt"

# Ver cambios preparados
git status

# Primer commit
git commit -m "Añadir archivo marcas tabaco.txt con listado de productos"

# Ver estado limpio
git status

# Ver historial
git log

# Ver archivos en repositorio
git ls-files



SEGUNDO COMMIT - ACTUALIZACIÓN

# Abrir VS Code en la carpeta
code .

# Ver cambios modificados
git status

# Añadir cambios al staging
git add "marcas tabaco.txt"

# Commit de actualización
git commit -m "Actualizar archivo marcas tabaco"

# Ver historial completo
git log


TRABAJAR CON RAMAS

# Ver ramas existentes
git branch

# Crear nueva rama
git branch nuevas-marcas-tabaco

# Cambiar a la nueva rama
git checkout nuevas-marcas-tabaco

# Ver ramas (con rama actual marcada)
git branch

# Forma moderna de crear y cambiar rama (nota)
git switch -c nuevas-marcas-tabaco


COMMIT EN NUEVA RAMA

# Ver estado en nueva rama
git status

# Añadir cambios
git add "marcas tabaco.txt"

# Commit en la rama nueva
git commit -m "añadir nuevas marcas de tabaco + japonesas"



VER HISTORIAL RESUMIDO

# Ver log en una línea
git log --oneline



FUSIÓN DE RAMAS (MERGE)

# Ver ramas disponibles
git branch

# Cambiar a la rama main
git checkout main

# Fusionar la rama nuevas-marcas-tabaco con main
git merge nuevas-marcas-tabaco

AÑADIR NUEVOS ARCHIVOS

# Ver archivos no rastreados
git status

# Añadir ambos archivos al staging
git add "Listado golosinas.txt" "Listado prensa.txt"

# Ver cambios preparados
git status

# Commit de los nuevos archivos
git commit -m "añadir listados de golosinas y prensa"

# Ver estado limpio
git status

# Ver historial reciente
git log --oneline -3



ELIMINAR ARCHIVO Y REVERTIR

# Eliminar archivo del repositorio
git rm "Listado prensa.txt"

# Ver cambios preparados
git status

# Commit de la eliminación
git commit -m "eliminar listado de prensa"

# Ver historial completo
git log --oneline

# Revertir un commit específico
git revert df6c977 --no-edit

# Verificar que el archivo volvió
ls -la



CONFIGURAR REPOSITORIO REMOTO

PRIMER PUSH A GITHUB

# Intentar push nuevamente (ya configurado)

# Verificar archivos descargados

SIMULAR CAMBIOS EN EL PROYECTO ORIGINAL

# Volver al proyecto original
cd /c/ESTANCOGIT

# Añadir cambio al archivo (usando echo)
echo "Revista Pronto" >> "Listado prensa.txt"

# Ver cambios no guardados
git status

# Preparar el cambio
git add "Listado prensa.txt"

# Hacer commit del cambio
git commit -m "Añadir revista Pronto"

# Ver estado (rama adelantada al remoto)
git status

# Subir cambios a GitHub
git push




GIT PULL - SINCRONIZAR LA COPIA

# Ir a la copia clonada
cd /c/Users/Mar_Bar/Desktop/ESTANCO-copia

# Descargar y fusionar cambios desde GitHub
git pull origin main

# Verificar que llegó el cambio
cat "Listado prensa.txt"








-------------

    git add - Preparar archivos nuevos

    git commit - Guardar cambios

    git rm - Eliminar archivos del repositorio

    git revert - Deshacer cambios de forma segura

    git remote add - Conectar con repositorio remoto

    git push -u origin main - Primer subida a GitHub
